                                                                                                                                        类作用场景
                                                                                                                                        Semaphore信号量，可以通过控制“许可证”的数量，保证线程之间的配合线程只有拿到“许可证”后才能继续操作
                                                                                                                                        CyclicBarrier线程会等待，直到足够多的线程达到了事先规定的数目。一旦到达触发条件，就可以进行下一步操作适用于线程之间互相等待处理结果就绪的场景
                                                                                                                                        Phaser与CyclicBarrier类似，但是计数器可变
                                                                                                                                        CountDownLatch与CyclicBarrier类似，数量减至0，触发动作不可以重复使用
                                                                                                                                        Exchanger让两个线程在合适的时候交换对象当一个线程作用于同一个类不同的实例上时用于交换数据
                                                                                                                                        Condition可以控制线程的“阻塞”和“唤醒”Object.wait()的升级版本

                                                                                                                                        # CountDownLatch

                                                                                                                                        # 作用

                                                                                                                                        开始-->线程进入等待-->倒数结束-->线程继续工作

                                                                                                                                        CountDownLatch不能重复使用。

                                                                                                                                        # 方法

                                                                                                                                        • 构造方法：CountDownLatch(int count)，参数count是需要倒数的值。
                                                                                                                                        • await()：调用CountDownLatch.await()的方法将被挂起，一直等到count编程0才执行。
                                                                                                                                        • countDown()：每次调用将count减1，直到变为0，线程被唤醒。

                                                                                                                                        # 常见实现

                                                                                                                                        # 一等多(参数为多)

                                                                                                                                        public class CountDownLatchTest1 {
    public static void main(String[] args) throws InterruptedException {
        CountDownLatch latch = new CountDownLatch(5);
        ExecutorService pool = Executors.newFixedThreadPool(5);
        for (int i = 0; i < 5; i++) {
            final int no = i+1;
            Runnable runnable = new Runnable() {
                @Override
                public void run() {
                    try {
                        Thread.sleep((long) (Math.random() * 10000));
                        System.out.println("No：" + no + "完成");
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    } finally {
                        latch.countDown();
                    }
                }
            };
            pool.submit(runnable);
        }
        System.out.println("等待5个线程");
        latch.await();
        System.out.println("所有线程检查完毕");
    }
}

                                                                                                                                        public class CountDownLatchTest2 {
    public static void main(String[] args) throws InterruptedException {
        CountDownLatch latch = new CountDownLatch(1);
        ExecutorService pool = Executors.newFixedThreadPool(5);
        for (int i = 0; i < 5; i++) {
            final int no = i + 1;
            Runnable runnable = new Runnable() {
                @Override
                public void run() {
                    System.out.println("线程"+no+"准备完毕");
                    try {
                        latch.await();
                        System.out.println("线程"+no+"执行完毕");
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
            };
            pool.submit(runnable);
        }
        System.out.println("主线程执行其他操作");
        Thread.sleep(5000);
        System.out.println("主线程完毕");
        latch.countDown();
    }
}

                                                                                                                                        # await()方法

                                                                                                                                        // 调用Sync的await()
public void await() throws InterruptedException {
    // 共享的且允许中断
    sync.acquireSharedInterruptibly(1);
}

public final void acquireSharedInterruptibly(int arg)
        throws InterruptedException {
    if (Thread.interrupted())
        throw new InterruptedException();
    // 满足条件入队进入等待队列
    if (tryAcquireShared(arg) < 0)
        doAcquireSharedInterruptibly(arg);
}
// 条件是state的值，如果所有的count到了0了，就可以执行，不需要进入队列阻塞等待
// 如果state不等于0说明还有count没有执行完成，线程进入队列
protected int tryAcquireShared(int acquires) {
    return (getState() == 0) ? 1 : -1;
}

private void doAcquireSharedInterruptibly(int arg)
    throws InterruptedException {
    // 将当前线程包装成Node节点
    final Node node = addWaiter(Node.SHARED);
    boolean failed = true;
    try {
        for (;;) {
            final Node p = node.predecessor();
            if (p == head) {
                int r = tryAcquireShared(arg);
                if (r >= 0) {
                    setHeadAndPropagate(node, r);
                    p.next = null; // help GC
                    failed = false;
                    return;
                }
            }
            if (shouldParkAfterFailedAcquire(p, node) &&
                //阻塞线程，调用Unsafe.park()方法挂起线程
                parkAndCheckInterrupt())
                throw new InterruptedException();
        }
    } finally {
        if (failed)
            cancelAcquire(node);
    }
}

                                                                                                                                        # CountDown()方法

                                                                                                                                        //调用realease相关方法
public void countDown() {
    sync.releaseShared(1);
}
public final boolean releaseShared(int arg) {
    // 判断是否能够释放
    if (tryReleaseShared(arg)) {
        // 唤醒所有线程  
        doReleaseShared();
        return true;
    }
    return false;
}

protected boolean tryReleaseShared(int releases) {
    // 通过自旋把state-1
    // 如果能够成功，则CAS更新-1后的值
    for (;;) {
        int c = getState();
        if (c == 0)
            return false;
        int nextc = c-1;
        if (compareAndSetState(c, nextc))
            // 最终count为0则允许唤醒
            return nextc == 0;
    }
}

                                                                                                                                        # Samphore

                                                                                                                                        # 作用

                                                                                                                                        信号量是用来限制和管理数量有限的资源的使用情况。许可证不能办法的过多，只能部分的线程拿到。

                                                                                                                                        # 方法

                                                                                                                                        * 构造方法：new Semaphore(int,boolean fair)，第一个参数设置信号量的数量，第二个参数设置是否公平。
* acquire()：获取信号量，可以响应中断，会阻塞等待获取
* tryAcquire(timeout)：尝试获取信号量，如果没有信号量不会阻塞，超时后继续执行其他流程
* acquireUninterruptibly()：获取信号量，不响应中断，会阻塞等待获取
* release()：归还许可

                                                                                                                                        # 源码分析

                                                                                                                                        Semaphore也是通过实现一个内部类Sync继承AQS来实现相关方法。state表示许可证数量

                                                                                                                                        # acquire()方法

                                                                                                                                        // 调用共享acquire方法
public void acquire() throws InterruptedException {
    sync.acquireSharedInterruptibly(1);
}
public final void acquireSharedInterruptibly(int arg)
        throws InterruptedException {
    if (Thread.interrupted())
        throw new InterruptedException();
    if (tryAcquireShared(arg) < 0)
        doAcquireSharedInterruptibly(arg);
}

// 根据公平和非公平有两种实现，下面演示非公平的实现
final int nonfairTryAcquireShared(int acquires) {
    for (;;) {
        // 获取state
        int available = getState();
        int remaining = available - acquires;
        // 判断是否能够获取许可证
        if (remaining < 0 ||
            compareAndSetState(available, remaining))
            return remaining;
    }
}

                                                                                                                                        # Condition接口

                                                                                                                                        # 作用

                                                                                                                                        当一个线程需要等待某个条件时，使用Condition.await()进行等待，此时线程阻塞；另一个去执行条件，如果条件达到调用Condition.single()，此时唤醒对应等待的线程执行。

                                                                                                                                        # 方法

                                                                                                                                        • await()：阻塞，等待唤醒条件
                                                                                                                                        • single()：唤醒等待时间最长的线程
                                                                                                                                        • single()：唤醒所有阻塞等待的线程

                                                                                                                                        # 常用实现

                                                                                                                                        public class ConditionTest {
    // condition 依赖于锁
    private ReentrantLock lock = new ReentrantLock();
    private Condition condition = lock.newCondition();
    void method1() {
        lock.lock();
        try {
            System.out.println("条件不满足，开始等待");
            condition.await();
            System.out.println("方法满足，开始执行");
        } catch (InterruptedException e) {
            e.printStackTrace();
        } finally {
            lock.unlock();
        }
    }
    void method2() {
        lock.lock();
        try {
            System.out.println("准备工作结束，唤醒其他线程");
            condition.signalAll();
        } finally {
            lock.unlock();
        }
    }
    public static void main(String[] args) {
        ConditionTest test = new ConditionTest();
        new Thread(() -> {
            try {
                Thread.sleep(5000);
                test.method2();
            } catch (InterruptedException e) {
                e.printStackTrace();
            }
        }).start();
        test.method1();
    }
 }

                                                                                                                                        # CyclicBarrier

                                                                                                                                        # 作用

                                                                                                                                        如果需要大量的线程相互配合，分别计算不同的任务，并且最后需要统一汇总，就可以使用CyclicBarrier。CyclicBarrier可以构造一个集结点，当线程执行完毕就会在集结点等待，直到所有的线程执行到集结点，所有的线程再统一出发。

                                                                                                                                        • 与CountDownLatch相比，它是线程驱动的
                                                                                                                                        • 与CountDownLatch相比，它可以重用

                                                                                                                                        # 实现

                                                                                                                                        public class CyclicBarrierTest {
    public static void main(String[] args) {
        // 需要设置有几个对象，同时还需要设置条件到达后需要做的任务
        CyclicBarrier barrier = new CyclicBarrier(5, () -> {
            System.out.println("所有线程都到场");
        });
        for (int i = 0; i < 5; i++) {
            new Thread(new Task(i+1,barrier)).start();
        }
    }
    static class Task implements Runnable {
        private int id;
        private CyclicBarrier cyclicBarrier;
        public Task(int id, CyclicBarrier cyclicBarrier) {
            this.id = id;
            this.cyclicBarrier = cyclicBarrier;
        }
        @Override
        public void run() {
            System.out.println("线程"+id+"准备开始");
            try {
                Thread.sleep((long) (Math.random() * 10000));
                System.out.println("线程"+id+"到达等待状态");
                cyclicBarrier.await();
                System.out.println("线程"+id+"执行其他逻辑");
            } catch (InterruptedException e) {
                e.printStackTrace();
            } catch (BrokenBarrierException e) {
                e.printStackTrace();
            }
        }
    }
}
