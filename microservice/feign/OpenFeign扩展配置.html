                                                                # yml配置列表

                                                                对于OpenFeign的配置，相关特殊需求开发的案例我们单独拿出来进行分析，这里只做比较常用的配置的讲解，一般情况下，我们很少配置契约的，因此这里只列出yml方式配置，其他常用的组件替换的配置在下面详细说明。

                                                                对于简单的配置，推荐使用yml进行配置。

                                                                feign:
  client:
    config:
      default:
        connectTimeout: 5000 # 连接超时时间
        readTimeout: 5000   #读取超时时间
        loggerLevel: BASIC   #日志级别
        errorDecoder: com.xxx.xxxDecoder   # 错误码解析器
        retryer: xxxx.xxxRetryer     # 重试策略
        requestIntercepter:         # 拦截器
          - xxx.xxxIntercepter
          - xxx.xxxIntercepter
        decode404: false  # 是否对 404 错误码解码，详见 feign.SynchronousMethodHandler#executeAndDecode
        encoder: xxx.xxxEncoder    # 编码器
        decoder: xxx.xxxEncoder    # 解码器
        contract: xxx.xxxContract  # 契约
  # 配置 GZIP 来压缩数据
  compression:
    request:
      enabled: true
      # 配置压缩的类型
      mime-types: text/xml,application/xml,application/json
      # 最小压缩值
      min-request-size: 2048
    response:
      enabled: true

                                                                OpenFeign的配置方式主要有两种，一种是通过yml直接进行配置；另一种是通过Java Bean的方式进行配置。

                                                                feign:
  client:
    config:
      default: # 默认则为全局配置
        loggerLevel: BASIC

                                                                如果是使用JavaBean的方式配置，则一定不要在配置类中使用@Configuration，而是在Feign接口的@FeignClient注解中配置自定义的配置类。

                                                                // 配置类不要使用@Configuration注入

    public Logger.Level feignLoggerLevel() {
        return Logger.Level.BASIC;
    }
}

// 在具体接口中配置上配置类
@FeignClient(value = "nacos-provider",path = "/nacos/provider",configuration = FeignConfig.class)
public interface CallNacosProviderFeign {
    @PostMapping("/simple")
    Position simpleProvider(@RequestBody Identity identity);
}

                                                                # Logger 日志级别配置

                                                                Feign对Logger提供了4中日志级别，分别是：

                                                                日志级别说明使用建议
                                                                NONE（默认）不打印日志不建议使用
                                                                BASIC记录请求方法、URL、响应状态码以及执行时间建议生产环境
                                                                HEADERS在 BASIC 的基础上，记录请求和响应的HEADERSdev 或 test 环境
                                                                FULL记录请求和响应的Header、Body和元数据dev 环境

                                                                一般情况下，对于接口调用日志的优化我们遵循这么几个原则：

                                                                • 生产环境下，如果不进行接口数据统计和收集，或者作用较小的接口，可以使用NONE级别的日志，在具体的调用逻辑的业务类上使用异常或者日志记录个整体调用情况即可。
                                                                • 生产环境下，如果需要统计和收集接口信息，或者数据组需要进行收集和分析的话，一般使用BASIC级别的日志足以，这个级别下可以记录到完整的请求URL、METHOD、STATUS、HEADERS。
                                                                • 对于测试环境或者生产环境下，如果需要对接口性能进行分析，则可以使用HEADERS或者FULL。 下面我们使用yml和Java Bean的方式分别配置不同级别的日志，使用yml配置日志的一个前提条件就是对应feign路径下的日志首先要打开开启到debug级别：
                                                                # yml 方式
logging:
  level:
    cn.xxx.demo.feign: debug
feign:
  client:
    config:
      nacos-provider: # 微服务名
        loggerLevel: BASIC

                                                                # 自定义 Interceptor 拦截器

                                                                自己的拦截器只要实现RequestInterceptor并且实现apply方法即可，apply方法可以获取到RequestTemplate，通过RequestTemplate可以对请求进行预处理，比如下面加入token值：

                                                                public class FeignInterceptor implements RequestInterceptor {
    @Override
    public void apply(RequestTemplate requestTemplate) {
        String token = UUID.randomUUID().toString();
        requestTemplate.header("Authorization",token);
    }
}

                                                                # HttpURLConnection 使用连接池

                                                                由于 Feign 默认的调用方式是 HttpURLConnection 并不是使用连接池，因此可以配置连接池来优化性能，实际调用效果约提升个 10%-15% 左右。

                                                                Feign 还支持阿帕奇的 HttpClient、OkHttp 发送请求，HttpClient 和 OkHttp 都是支持连接池的。

                                                                • 添加 Feign-httpClient 依赖
                                                                <dependency>
    <groupId>io.github.openfeign</groupId>
    <artifactId>feign-httpclient</artifactId>
</dependency>
                                                                1
                                                                2
                                                                3
                                                                4
                                                                • 配置文件
                                                                feign:
  httpclient:
    enabled: true   #使用 httpClient
    max-connections: 200    # 最大连接数
    max-connections-per-route: 50   # 单个路径最大连接数
                                                                1
                                                                2
                                                                3
                                                                4
                                                                5

                                                                可以根据压测结果配置一个连接数和单路径连接数的优化值。
