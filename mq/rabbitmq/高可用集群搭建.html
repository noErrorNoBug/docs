                                        # 创建三个节点

                                        创建三个节点时，注意使用相同的cookie以便建立集群。

                                        #################################
#RabibitMQ Node1
#################################
docker run 
-d --name=rabbitmq-node-1
-p 5671:5672 
-p 15671:15672 
-v /mnt/rabbitmq-clusters/data/node1:/var/lib/rabbitmq:z 
-e RABBITMQ_DEFAULT_USER=admin 
-e RABBITMQ_DEFAULT_PASS=rabbitmq 
-e RABBITMQ_NODENAME=rabbitmq-node-1
-e RABBITMQ_ERLANG_COOKIE='rabbitMQ-erlang-cluster-cookie' 
-h rabbitmq-node-1 
--network=rabbitmq-net 
rabbitmq:3.8.14-management
 
#################################
#RabibitMQ Node2

                                        # 构建集群并修改为镜像策略

                                        组件RabbitMQ集群，将Node-2和Node-3加入到Node-1中

                                        #################################
#RabibitMQ Node2
#################################
docker exec -it rabbitmq-node-2 bash
# 操作Node2
rabbitmqctl stop_app
rabbitmqctl reset
rabbitmqctl join_cluster rabbitmq-node-1@rabbitmq-node-1
rabbitmqctl start_app
exit

#################################
#RabibitMQ Node3
#################################
docker exec -it rabbitmq-node-3 bash
# 操作Node2
rabbitmqctl stop_app
rabbitmqctl reset
rabbitmqctl join_cluster rabbitmq-node-1@rabbitmq-node-1
rabbitmqctl start_app
exit

                                        # 安装配置HaProxy

                                        #拉取镜像
docker pull haproxy:2.3.9
                                        1
                                        2

                                        配置文件如下所示

                                        # 全局配置
global
    log 127.0.0.1 local2
    daemon
    maxconn 4000
# 默认配置
defaults
    mode http
    log global
    option httplog
    option dontlognull
    timeout connect 5000ms
    timeout client 1m
    timeout server 1m
# 监听 RabbitMQ 集群
listen rabbitmq_cluster
    # 通过5671端口对下面三个server进行映射
    bind 0.0.0.0:5670
    # 记录TCP连接时间和状态日志
    option tcplog
    # TCP协议代理转发
    mode tcp
    # 开启TCP的长连接模式 Keep alive
    option clitcpka
    # 负载均衡策略使用轮询(roundrobin)
    balance roundrobin
    # HaProxy 与 RabbitMQ 建立连接的超时时间
    timeout connect 1s
    # 客户端与 HaProxy 的最大空闲时间
    timeout client 10s
    # 服务器与 HaProxy 的最大空闲时间
    timeout server 10s
    # 每隔5s发送一次心跳包,如果连续两次心跳良好认定服务健康；如果连续三次失败认为服务宕机剔除
    server rabbitmq-node-1 sunqiang.club:5671 check inter 2s rise 2 fall 3
    server rabbitmq-node-2 sunqiang.club:5672 check inter 2s rise 2 fall 3
    server rabbitmq-node-3 sunqiang.club:5673 check inter 2s rise 2 fall 3
# 开启HaProxy监控服务
listen http_front
    # 暴露端口
    bind 0.0.0.0:15674
    # 统计页面自动刷新时间
    stats refresh 30s
    # 统计页面URL
    stats uri /haproxy?stats
    # 统计页面用户名密码
    stats auth admin:haproxy
# 监听 RabbitMQ web页面集群
listen rabbitmq_admin
    # 绑定端口
    bind 0.0.0.0:15670
    # 绑定三个web页面
    server rabbitmq-node-1 sunqiang.club:15671 check inter 2s rise 2 fall 3
    server rabbitmq-node-2 sunqiang.club:15672 check inter 2s rise 2 fall 3
    server rabbitmq-node-3 sunqiang.club:15673 check inter 2s rise 2 fall 3
